' == Activity: Login ==
@startuml
title Activity Diagram - Login Enerlyze
|Pengguna|
start
:Buka halaman Login Enerlyze;
:Masukkan email dan password;
:Tekan tombol Login;
|Sistem|
:Verifikasi email & password
terhadap data pengguna;
if (Kredensial valid?) then (ya)
  :Buat sesi login &
  segarkan notifikasi;
  if (Role pengguna?) then (Admin)
    :Arahkan ke Dashboard Admin;
  else (Staff)
    :Arahkan ke Dashboard Staff;
  endif
  stop
else (tidak)
  :Tampilkan pesan
  "Email atau password salah";
  stop
endif
@enduml

|Staff|
start
:Buka halaman Data;
:Pilih area & bulan;

  :Isi form data energi
  (kWh, biaya, durasi, maintenance);
  :Tekan Simpan;
  |Sistem|
  if (Data area & bulan sudah ada?) then (ya, saat Tambah)
    :Tampilkan pesan data sudah ada;
    stop

    :Tampilkan notifikasi berhasil;
    stop
  endif
else (Hapus)
  |Sistem|
  :Konfirmasi & hapus catatan data;
  :Tampilkan notifikasi berhasil;
  stop
endif
@enduml

' == Activity: Jalankan Analisis AHP Bulanan ==
@startuml
title Activity Diagram - Jalankan Analisis AHP Bulanan
|Admin|
start
:Buka halaman Data Admin;
:Pilih bulan analisis;
|Sistem|
:Periksa kelengkapan data
seluruh area pada bulan tsb;
if (Semua area sudah mengisi data?) then (ya)
  |Admin|
  :Tekan Run / Re-run AHP &
  konfirmasi;
  |Sistem|
  :Hitung bobot kriteria, konsistensi
  & ranking prioritas area;
  :Simpan hasil AHP bulan tsb;
  :Tampilkan ranking & notifikasi berhasil;
  stop
else (tidak)
  :Nonaktifkan tombol Run AHP &
  tampilkan info data belum lengkap;
  stop
endif
@enduml
